# Para usar Gmail SMTP, reemplaza estas líneas en tu .env:

MAIL_MAILER=smtp
MAIL_HOST=smtp.gmail.com
MAIL_PORT=587
MAIL_USERNAME=tu-email@gmail.com
MAIL_PASSWORD=tu-app-password  # NO tu contraseña normal
MAIL_ENCRYPTION=tls
MAIL_FROM_ADDRESS="tu-email@gmail.com"
MAIL_FROM_NAME="Sistema de Seguridad"

# IMPORTANTE: 
# 1. Activa "Verificación en 2 pasos" en tu cuenta Gmail
# 2. Genera una "Contraseña de aplicación" específica
# 3. Usa esa contraseña de aplicación, NO tu contraseña normal
